Nižšie máš **kompletný `sdram_tb.sv` testbench**, ktorý je kompatibilný s tvojím **Makefile frameworkom**. Je navrhnutý tak, aby:

* generoval **clock + reset**
* obsahoval **jednoduchý SDRAM behavioral model**
* stimuloval **AXI read transakcie**
* bol kompatibilný s **ModelSim / QuestaSim**
* umožňoval jednoduché **wave debugging**

Testbench je štýlom podobný tomu, ktorý používajú open-source projekty a aj referenčné simulácie od Intel a AMD.

# `tb/sdram_tb.sv`

```systemverilog
/**
 * @file sdram_tb.sv
 * @brief Testbench pre SDRAM controller subsystem
 *
 * Testuje:
 *  - SDRAM inicializáciu
 *  - AXI read request
 *  - scheduler command generation
 *  - PHY command interface
 *
 * Používa jednoduchý behavioral SDRAM model.
 */

`timescale 1ns/1ps
`default_nettype none

import sdram_pkg::*;

module sdram_tb;

  // ==========================================================
  // CLOCK + RESET
  // ==========================================================

  logic clk;
  logic clk_sh;
  logic rstn;

  initial clk = 0;
  always #5 clk = ~clk;        // 100 MHz

  initial clk_sh = 0;
  always #5 clk_sh = ~clk_sh;  // jednoduchý model (reálne by bol phase shift)

  initial begin
    rstn = 0;
    #100;
    rstn = 1;
  end

  // ==========================================================
  // AXI INTERFACE SIGNALS
  // ==========================================================

  logic [3:0]  s_axi_arid;
  logic [31:0] s_axi_araddr;
  logic [7:0]  s_axi_arlen;
  logic        s_axi_arvalid;
  logic        s_axi_arready;

  logic [3:0]  s_axi_rid;
  logic [31:0] s_axi_rdata;
  logic        s_axi_rlast;
  logic        s_axi_rvalid;
  logic        s_axi_rready;

  // ==========================================================
  // SDRAM PHYSICAL PINS
  // ==========================================================

  logic [12:0] sdram_addr;
  logic [1:0]  sdram_ba;

  logic sdram_ras_n;
  logic sdram_cas_n;
  logic sdram_we_n;
  logic sdram_cs_n;

  logic sdram_clk;
  logic sdram_cke;

  wire [15:0] sdram_dq;

  // ==========================================================
  // DUT
  // ==========================================================

  sdram_system_top dut (

    .clk(clk),
    .clk_sh(clk_sh),
    .rstn(rstn),

    .s_axi_arid(s_axi_arid),
    .s_axi_araddr(s_axi_araddr),
    .s_axi_arlen(s_axi_arlen),
    .s_axi_arvalid(s_axi_arvalid),
    .s_axi_arready(s_axi_arready),

    .s_axi_rid(s_axi_rid),
    .s_axi_rdata(s_axi_rdata),
    .s_axi_rlast(s_axi_rlast),
    .s_axi_rvalid(s_axi_rvalid),
    .s_axi_rready(s_axi_rready),

    .sdram_addr(sdram_addr),
    .sdram_ba(sdram_ba),

    .sdram_ras_n(sdram_ras_n),
    .sdram_cas_n(sdram_cas_n),
    .sdram_we_n(sdram_we_n),
    .sdram_cs_n(sdram_cs_n),

    .sdram_dq(sdram_dq),

    .sdram_clk(sdram_clk),
    .sdram_cke(sdram_cke)

  );

  // ==========================================================
  // SIMPLE SDRAM MODEL
  // ==========================================================

  sdram_model mem (

    .clk(sdram_clk),

    .addr(sdram_addr),
    .ba(sdram_ba),

    .ras_n(sdram_ras_n),
    .cas_n(sdram_cas_n),
    .we_n(sdram_we_n),
    .cs_n(sdram_cs_n),

    .dq(sdram_dq)

  );

  // ==========================================================
  // AXI DRIVER
  // ==========================================================

  task axi_read(input logic [31:0] addr);

    begin

      @(posedge clk);

      s_axi_araddr  <= addr;
      s_axi_arlen   <= 0;
      s_axi_arid    <= 0;
      s_axi_arvalid <= 1;

      wait(s_axi_arready);

      @(posedge clk);

      s_axi_arvalid <= 0;

    end

  endtask

  // ==========================================================
  // TEST SEQUENCE
  // ==========================================================

  initial begin

    s_axi_arvalid = 0;
    s_axi_rready  = 1;

    wait(rstn);

    // počkaj na SDRAM init
    #300000;

    $display("Issuing AXI reads");

    axi_read(32'h00001000);
    axi_read(32'h00002000);
    axi_read(32'h00003000);

    #100000;

    $display("Simulation finished");

    $finish;

  end

endmodule
```

# Jednoduchý SDRAM model

Vytvor `rtl/sdram_model.sv`.

```systemverilog
/**
 * Very simple behavioral SDR SDRAM model
 *
 * Nie je timing-accurate, ale postačuje na
 * kontrolu command sekvencií.
 */

`default_nettype none

module sdram_model(

  input logic clk,

  input logic [12:0] addr,
  input logic [1:0]  ba,

  input logic ras_n,
  input logic cas_n,
  input logic we_n,
  input logic cs_n,

  inout wire [15:0] dq

);

  logic [15:0] mem [0:65535];

  logic [15:0] dq_out;
  logic dq_drive;

  assign dq = dq_drive ? dq_out : 'z;

  always_ff @(posedge clk) begin

    dq_drive <= 0;

    if(!cs_n) begin

      // READ
      if(!ras_n && !cas_n && we_n) begin
        dq_out <= mem[addr];
        dq_drive <= 1;
      end

      // WRITE
      if(!ras_n && !cas_n && !we_n) begin
        mem[addr] <= dq;
      end

    end

  end

endmodule
```

# Odporúčané wave signály

Ak chceš debugging podobný tomu čo používaš pri XFCP:

```tcl
add wave -divider "AXI"
add wave sim:/sdram_tb/s_axi_*

add wave -divider "SDRAM COMMAND"
add wave sim:/sdram_tb/sdram_ras_n
add wave sim:/sdram_tb/sdram_cas_n
add wave sim:/sdram_tb/sdram_we_n
add wave sim:/sdram_tb/sdram_cs_n

add wave -divider "ADDRESS"
add wave -hex sim:/sdram_tb/sdram_addr
add wave -hex sim:/sdram_tb/sdram_ba

add wave -divider "DUT INTERNAL"
add wave sim:/sdram_tb/dut/sched_cmd
add wave sim:/sdram_tb/dut/init_done

run -all
```
