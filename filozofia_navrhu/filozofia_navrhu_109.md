## `socfw/emit/board_tcl_emitter.py`

```python
from __future__ import annotations

from pathlib import Path


class BoardTclEmitter:
    def emit(self, *, out_dir: str, system) -> str:
        hal_dir = Path(out_dir) / "hal"
        hal_dir.mkdir(parents=True, exist_ok=True)

        out = hal_dir / "board.tcl"

        lines: list[str] = []
        lines.append("# Generated by socfw")
        lines.append("")

        self._emit_device(lines, system.board)
        self._emit_system_pins(lines, system.board)
        self._emit_resources(lines, system.board.resources)

        out.write_text("\n".join(lines).rstrip() + "\n", encoding="utf-8")
        return str(out)

    def _emit_device(self, lines: list[str], board) -> None:
        if board.fpga_part:
            lines.append("# FPGA device")
            lines.append(f"set_global_assignment -name DEVICE {board.fpga_part}")
            lines.append("")

        if board.fpga_family:
            lines.append("# FPGA family")
            lines.append(f"# family: {board.fpga_family}")
            lines.append("")

    def _emit_system_pins(self, lines: list[str], board) -> None:
        lines.append("# System pins")
        clk = board.system_clock
        lines.append(f"set_location_assignment {clk.pin} -to {clk.top_name}")

        if board.system_reset is not None:
            rst = board.system_reset
            lines.append(f"set_location_assignment {rst.pin} -to {rst.top_name}")

        lines.append("")

    def _emit_resources(self, lines: list[str], resources: dict) -> None:
        flat = []

        def walk(node, path: str):
            if not isinstance(node, dict):
                return

            if "kind" in node and "top_name" in node:
                flat.append((path, node))
                return

            for key in sorted(node):
                child = node[key]
                child_path = f"{path}.{key}" if path else key
                walk(child, child_path)

        walk(resources, "")

        if not flat:
            return

        lines.append("# Board resources")
        for path, res in sorted(flat, key=lambda x: x[0]):
            kind = res.get("kind")
            top_name = res.get("top_name")

            if kind == "scalar":
                pin = res.get("pin")
                if pin:
                    lines.append(f"# {path}")
                    lines.append(f"set_location_assignment {pin} -to {top_name}")

            elif kind in {"vector", "inout"}:
                pins = list(res.get("pins", []))
                for idx, pin in enumerate(pins):
                    lines.append(f"set_location_assignment {pin} -to {top_name}[{idx}]")

        lines.append("")
```

## `tests/unit/test_board_tcl_emitter.py`

```python
from pathlib import Path

from socfw.emit.board_tcl_emitter import BoardTclEmitter
from socfw.model.board import BoardClock, BoardModel, BoardReset
from socfw.model.project import ProjectModel
from socfw.model.system import SystemModel


def test_board_tcl_emitter_writes_clock_reset_and_leds(tmp_path):
    board = BoardModel(
        board_id="demo",
        fpga_family="cyclone_iv_e",
        fpga_part="EP4CE55F23C8",
        system_clock=BoardClock(
            id="sys_clk",
            top_name="SYS_CLK",
            pin="PIN_A1",
            frequency_hz=50_000_000,
        ),
        system_reset=BoardReset(
            id="reset_n",
            top_name="RESET_N",
            pin="PIN_B1",
            active_low=True,
        ),
        resources={
            "onboard": {
                "leds": {
                    "kind": "vector",
                    "top_name": "ONB_LEDS",
                    "width": 2,
                    "pins": ["PIN_C1", "PIN_C2"],
                }
            }
        },
    )

    system = SystemModel(
        board=board,
        project=ProjectModel(name="demo", mode="standalone", board_ref="demo"),
    )

    out = BoardTclEmitter().emit(out_dir=str(tmp_path), system=system)
    text = Path(out).read_text(encoding="utf-8")

    assert "set_global_assignment -name DEVICE EP4CE55F23C8" in text
    assert "set_location_assignment PIN_A1 -to SYS_CLK" in text
    assert "set_location_assignment PIN_B1 -to RESET_N" in text
    assert "set_location_assignment PIN_C1 -to ONB_LEDS[0]" in text
```
