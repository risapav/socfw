## `socfw/emit/sdc_emitter.py`

```python
from __future__ import annotations

from pathlib import Path


class SdcEmitter:
    def emit(self, *, out_dir: str, system) -> str:
        timing_dir = Path(out_dir) / "timing"
        timing_dir.mkdir(parents=True, exist_ok=True)

        out = timing_dir / "soc_top.sdc"
        lines: list[str] = []

        lines.append("# Generated by socfw")
        lines.append("")

        self._emit_primary_clock(lines, system)
        self._emit_generated_clocks(lines, system)
        self._emit_io_delays(lines, system)
        self._emit_false_paths(lines, system)

        out.write_text("\n".join(lines).rstrip() + "\n", encoding="utf-8")
        return str(out)

    def _emit_primary_clock(self, lines: list[str], system) -> None:
        clk = system.board.system_clock
        period_ns = 1_000_000_000.0 / float(clk.frequency_hz)

        lines.append("# Primary clocks")
        lines.append(
            f"create_clock -name {clk.top_name} "
            f"-period {period_ns:.3f} [get_ports {{{clk.top_name}}}]"
        )
        lines.append("")

    def _emit_generated_clocks(self, lines: list[str], system) -> None:
        timing = system.timing
        if timing is None or not timing.generated_clocks:
            return

        lines.append("# Generated clocks")
        for g in sorted(timing.generated_clocks, key=lambda x: x.name):
            args = []
            if g.divide_by is not None:
                args.append(f"-divide_by {g.divide_by}")
            if g.multiply_by is not None:
                args.append(f"-multiply_by {g.multiply_by}")

            arg_text = " ".join(args)
            if arg_text:
                arg_text = " " + arg_text

            lines.append(
                f"create_generated_clock -name {g.name}{arg_text} "
                f"-source [get_pins {{{g.source}}}] "
                f"[get_pins {{{g.target}}}]"
            )
        lines.append("")

    def _emit_io_delays(self, lines: list[str], system) -> None:
        timing = system.timing
        io = getattr(timing, "io_delays", None) if timing is not None else None
        if not isinstance(io, dict):
            return

        if not io.get("auto", False):
            return

        clock = io.get("clock") or system.board.system_clock.top_name
        input_max = io.get("default_input_max_ns")
        output_max = io.get("default_output_max_ns")

        if input_max is None and output_max is None:
            return

        lines.append("# Default IO delays")

        if input_max is not None:
            lines.append(
                f"set_input_delay -clock {clock} "
                f"-max {float(input_max):.3f} [all_inputs]"
            )

        if output_max is not None:
            lines.append(
                f"set_output_delay -clock {clock} "
                f"-max {float(output_max):.3f} [all_outputs]"
            )

        lines.append("")

    def _emit_false_paths(self, lines: list[str], system) -> None:
        timing = system.timing
        if timing is None or not timing.false_paths:
            return

        lines.append("# False paths")
        for fp in sorted(timing.false_paths, key=lambda x: (getattr(x, "from_path", ""), getattr(x, "to_path", ""))):
            from_path = getattr(fp, "from_path", None)
            to_path = getattr(fp, "to_path", None)

            if from_path and to_path:
                lines.append(f"set_false_path -from {from_path} -to {to_path}")
            elif from_path:
                lines.append(f"set_false_path -from [get_ports {{{from_path}}}]")
            elif hasattr(fp, "from_port"):
                lines.append(f"set_false_path -from [get_ports {{{fp.from_port}}}]")

        lines.append("")
```

## Potrebný malý model patch

Ak `TimingModel` ešte nemá `io_delays`, doplň:

```python
@dataclass
class TimingModel:
    clocks: list = field(default_factory=list)
    generated_clocks: list[GeneratedClock] = field(default_factory=list)
    io_delays: dict = field(default_factory=dict)
    false_paths: list[FalsePath] = field(default_factory=list)
    raw: dict = field(default_factory=dict)
```

A v `TimingLoader` prenes:

```python
io_delays=dict(doc.timing.io_delays or {})
```

podľa tvojho aktuálneho schema tvaru.

## `tests/unit/test_sdc_emitter.py`

```python
from pathlib import Path

from socfw.emit.sdc_emitter import SdcEmitter
from socfw.model.board import BoardClock, BoardModel
from socfw.model.system import SystemModel
from socfw.model.project import ProjectModel


def test_sdc_emitter_writes_primary_clock(tmp_path):
    system = SystemModel(
        board=BoardModel(
            board_id="demo",
            system_clock=BoardClock(
                id="sys_clk",
                top_name="SYS_CLK",
                pin="A1",
                frequency_hz=50_000_000,
            ),
        ),
        project=ProjectModel(
            name="demo",
            mode="standalone",
            board_ref="demo",
        ),
    )

    out = SdcEmitter().emit(out_dir=str(tmp_path), system=system)
    text = Path(out).read_text(encoding="utf-8")

    assert "create_clock" in text
    assert "-period 20.000" in text
    assert "SYS_CLK" in text
```
