## `socfw/emit/files_tcl_emitter.py`

```python
from __future__ import annotations

from pathlib import Path

from socfw.build.vendor_artifacts import collect_vendor_artifacts


class FilesTclEmitter:
    def emit(self, *, out_dir: str, system, planned_bridges: list) -> str:
        hal_dir = Path(out_dir) / "hal"
        hal_dir.mkdir(parents=True, exist_ok=True)

        out = hal_dir / "files.tcl"
        lines: list[str] = []

        lines.append("# Generated by socfw")
        lines.append("")

        for fp in self._collect_rtl_files(system, planned_bridges):
            lines.append(f"set_global_assignment -name VERILOG_FILE {fp}")

        vendor = collect_vendor_artifacts(system)

        if vendor.qip_files:
            lines.append("")
            lines.append("# Vendor QIP files")
            for qip in sorted(vendor.qip_files):
                lines.append(f"set_global_assignment -name QIP_FILE {qip}")

        if vendor.sdc_files:
            lines.append("")
            lines.append("# Vendor SDC files")
            for sdc in sorted(vendor.sdc_files):
                lines.append(f"set_global_assignment -name SDC_FILE {sdc}")

        out.write_text("\n".join(lines).rstrip() + "\n", encoding="utf-8")
        return str(out)

    def _collect_rtl_files(self, system, planned_bridges: list) -> list[str]:
        files: list[str] = []

        used_types = {m.type_name for m in system.project.modules}
        for type_name in sorted(used_types):
            ip = system.ip_catalog.get(type_name)
            if ip is None:
                continue

            for fp in getattr(ip, "synthesis_files", ()):
                # qip files should be emitted as QIP_FILE, not VERILOG_FILE
                if str(fp).endswith(".qip"):
                    continue
                files.append(str(fp))

        for bridge in planned_bridges:
            files.append(str(bridge.rtl_file))

        return sorted(dict.fromkeys(files))
```

## `tests/unit/test_files_tcl_emitter.py`

```python
from pathlib import Path
from dataclasses import dataclass

from socfw.emit.files_tcl_emitter import FilesTclEmitter
from socfw.model.board import BoardClock, BoardModel
from socfw.model.ip import IpDescriptor
from socfw.model.project import ProjectModel, ProjectModule
from socfw.model.system import SystemModel
from socfw.model.vendor import VendorInfo


@dataclass(frozen=True)
class DummyBridge:
    rtl_file: str


def test_files_tcl_emitter_writes_vendor_and_bridge_files(tmp_path):
    bridge_file = tmp_path / "bridge.sv"
    bridge_file.write_text("// bridge\n", encoding="utf-8")

    system = SystemModel(
        board=BoardModel(
            board_id="demo",
            system_clock=BoardClock(
                id="clk",
                top_name="SYS_CLK",
                pin="A1",
                frequency_hz=50_000_000,
            ),
        ),
        project=ProjectModel(
            name="demo",
            mode="standalone",
            board_ref="demo",
            modules=[ProjectModule(instance="pll0", type_name="clkpll")],
        ),
        ip_catalog={
            "clkpll": IpDescriptor(
                name="clkpll",
                module="clkpll",
                category="clocking",
                synthesis_files=("/tmp/clkpll.v", "/tmp/clkpll.qip"),
                vendor_info=VendorInfo(
                    vendor="intel",
                    tool="quartus",
                    qip="/tmp/clkpll.qip",
                    sdc=("/tmp/clkpll.sdc",),
                ),
            )
        },
    )

    out = FilesTclEmitter().emit(
        out_dir=str(tmp_path),
        system=system,
        planned_bridges=[DummyBridge(rtl_file=str(bridge_file))],
    )

    text = Path(out).read_text(encoding="utf-8")
    assert "VERILOG_FILE /tmp/clkpll.v" in text
    assert "VERILOG_FILE" in text
    assert "bridge.sv" in text
    assert "QIP_FILE /tmp/clkpll.qip" in text
    assert "SDC_FILE /tmp/clkpll.sdc" in text
```
